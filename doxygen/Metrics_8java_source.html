230  public boolean start() {
231  synchronized (optOutLock) {
232  // Did we opt out?
233  if (isOptOut()) {
234  return false;
235  }
236 
237  // Is metrics already running?
238  if (taskId >= 0) {
239  return true;
240  }
241 
242  // Begin hitting the server with glorious data
243  taskId = plugin.getEngine().getScheduler().scheduleAsyncRepeatingTask(plugin, new Runnable() {
244 
245  private boolean firstPost = true;
246 
247  public void run() {
248  try {
249  // This has to be synchronized or it can collide with the disable method.
250  synchronized (optOutLock) {
251  // Disable Task, if it is running and the server owner decided to opt-out
252  if (isOptOut() && taskId > 0) {
253  plugin.getEngine().getScheduler().cancelTask(taskId);
254  taskId = -1;
255  // Tell all plotters to stop gathering information.
256  for (Graph graph : graphs) {
257  graph.onOptOut();
258  }
259  }
260  }
261 
262  // We use the inverse of firstPost because if it is the first time we are posting,
263  // it is not a interval ping, so it evaluates to FALSE
264  // Each time thereafter it will evaluate to TRUE, i.e PING!
265  postPlugin(!firstPost);
266 
267  // After the first post we set firstPost to false
268  // Each post thereafter will be a ping
269  firstPost = false;
270  } catch (IOException e) {
271  Spout.getLogger().log(Level.INFO, "[Metrics] " + e.getMessage());
272  }
273  }
274  }, 0, PING_INTERVAL * 1200, TaskPriority.LOW);
275 
276  return true;
277  }

352  private void postPlugin(final boolean isPing) throws IOException {
353  // The plugin's description file containing all of the plugin data such as name, version, author, etc
354  final PluginDescriptionFile description = plugin.getDescription();
355 
356  // Construct the post data
357  final StringBuilder data = new StringBuilder();
358  data.append(encode("guid")).append('=').append(encode(guid));
359  encodeDataPair(data, "version", description.getVersion());
360  encodeDataPair(data, "server", Spout.getAPIVersion());
361  encodeDataPair(data, "players", Integer.toString(Spout.getEngine().getOnlinePlayers().length));
362  encodeDataPair(data, "revision", String.valueOf(REVISION));
363 
364  // If we're pinging, append it
365  if (isPing) {
366  encodeDataPair(data, "ping", "true");
367  }
368 
369  // Acquire a lock on the graphs, which lets us make the assumption we also lock everything
370  // inside of the graph (e.g plotters)
371  synchronized (graphs) {
372  final Iterator<Graph> iter = graphs.iterator();
373 
374  while (iter.hasNext()) {
375  final Graph graph = iter.next();
376 
377  for (Plotter plotter : graph.getPlotters()) {
378  // The key name to send to the metrics server
379  // The format is C-GRAPHNAME-PLOTTERNAME where separator - is defined at the top
380  // Legacy (R4) submitters use the format Custom%s, or CustomPLOTTERNAME
381  final String key = String.format("C%s%s%s%s", CUSTOM_DATA_SEPARATOR, graph.getName(), CUSTOM_DATA_SEPARATOR, plotter.getColumnName());
382 
383  // The value to send, which for the foreseeable future is just the string
384  // value of plotter.getValue()
385  final String value = Integer.toString(plotter.getValue());
386 
387  // Add it to the http post data :)
388  encodeDataPair(data, key, value);
389  }
390  }
391  }
392 
393  // Create the url
394  URL url = new URL(BASE_URL + String.format(REPORT_URL, encode(plugin.getDescription().getName())));
395 
396  // Connect to the website
397  URLConnection connection;
398 
399  // Mineshafter creates a socks proxy, so we can safely bypass it
400  // It does not reroute POST requests so we need to go around it
401  if (isMineshafterPresent()) {
402  connection = url.openConnection(Proxy.NO_PROXY);
403  } else {
404  connection = url.openConnection();
405  }
406 
407  connection.setDoOutput(true);
408 
409  // Write the data
410  final OutputStreamWriter writer = new OutputStreamWriter(connection.getOutputStream());
411  writer.write(data.toString());
412  writer.flush();
413 
414  // Now read the response
415  final BufferedReader reader = new BufferedReader(new InputStreamReader(connection.getInputStream()));
416  final String response = reader.readLine();
417 
418  // close resources
419  writer.close();
420  reader.close();
421 
422  if (response == null || response.startsWith("ERR")) {
423  throw new IOException(response); //Throw the exception
424  } else {
425  // Is this the first update this hour?
426  if (response.contains("OK This is your first update this hour")) {
427  synchronized (graphs) {
428  final Iterator<Graph> iter = graphs.iterator();
429 
430  while (iter.hasNext()) {
431  final Graph graph = iter.next();
432 
433  for (Plotter plotter : graph.getPlotters()) {
434  plotter.reset();
435  }
436  }
437  }
438  }
439  }
